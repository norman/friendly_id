Examples:

def exists?(id = false)
  return super if id.unfriendly_id?
  super @klass.friendly_id_config.query_field => id
end

def find_one(id)
  return super if id.unfriendly_id?
  where(@klass.friendly_id_config.query_field => id).first or super
end
